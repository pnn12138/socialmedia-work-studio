## 自动加载机制

所有脚本已配置为**自动查找并加载根目录的 `.env` 文件**，无需手动指定路径。

### Node.js 脚本（image-finder）

```javascript
// .claude/skills/image-finder/index.js
const path = require('path');
const dotenv = require('dotenv');

function findRootEnv() {
  let currentDir = __dirname;
  for (let i = 0; i < 5; i++) {
    const envPath = path.join(currentDir, '.env');
    if (require('fs').existsSync(envPath)) {
      return envPath;
    }
    currentDir = path.dirname(currentDir);
  }
  return null;
}

const envPath = findRootEnv();
if (envPath) {
  dotenv.config({ path: envPath });
}
```

### Python 脚本（image-compose, smart-cropper, image-builder）

```python
# .claude/skills/{skill-name}/*.py
from dotenv import load_dotenv
from pathlib import Path

def find_root_env():
    """向上查找根目录的.env 文件"""
    current = Path(__file__).resolve()
    for _ in range(5):
        current = current.parent
        env_path = current / '.env'
        if env_path.exists():
            return env_path
    return None

root_env = find_root_env()
if root_env:
    load_dotenv(dotenv_path=root_env)
```
